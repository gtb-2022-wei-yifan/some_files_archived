function odd-or-even(){
    num=$1
    result=$(expr $num % 2 2>/dev/null)
    if [[ $result == 0 ]]
    then 
        echo "The num is even"
    elif 
        [[ $result == 1 ]]
    then
        echo "The num is odd"
    else
        echo "This is not an integer"
    fi
